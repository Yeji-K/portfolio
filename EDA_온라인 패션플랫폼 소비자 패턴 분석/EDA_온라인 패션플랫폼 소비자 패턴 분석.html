def make_generation(age):
    if age == -1:
        return '미입력'
    elif age // 10 >= 4:
        return "30대 후반"
    elif age // 10 == 1:
        return "10대"
    elif age % 10 < 3:
        return str(age // 10 * 10) + f"대 초반"
    elif age % 10 <= 6:
        return str(age // 10 * 10) + f"대 중반"
    else:
        return str(age // 10 * 10) + f"대 후반"

def encode_style_list(df):
    for i in df.index :       
        value = df.loc[i,"style"]
        if pd.notnull(value) :
            styles = value.split("/")
            for style in styles :
                df.loc[i,style] = True
    return df
